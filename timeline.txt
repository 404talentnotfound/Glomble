    .importance:
        "!" - vital
        "*" - important
        "&" - unimportant

    .progress:
        "-" - not yet started
        "|" - in progress
        "=" - finished

    .difficulty:
        "+" - difficult
        "$" - standard
        "#" - easy

    .misc:
        "~" - category
        "." - subcategory
        "/" - commentary

    .order:
        importance - progress - difficulty
